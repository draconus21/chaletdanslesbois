\mysection{Essential Amenities and Safety}

\textbf{Septic Tank}
\begin{itemize}
    \item Since all wastewater goes into a sealed septic tank, please be mindful of your water use.
    \item A red indicator light near the hot tub will illuminate when the septic tank reaches 80\% capacity
    \item Please notify us immediately if you notice this warning light is on
\end{itemize}

\textbf{Hot Tub}
\begin{itemize}
    \item Set to 99°F upon arrival (takes a few hours to heat up from 80°F)
    \item Keep cover on when not in use
    \item No glass containers - use provided plastic cups
    \item \textbf{Children require adult supervision at all times}
\end{itemize}

\textbf{Indoor Fireplace}
\begin{itemize}

    \item \textbf{Always close door and airflow before leaving/sleeping}

\begin{itemize}
    \item Check \href{https://sopfeu.qc.ca}{SOPFEU website} for fire bans first

    \item \textbf{Extinguish completely before going inside}

\mysection{Departure Checklist}
\textbf{Our checkout process is simple - Please leave the chalet as you found it, with these additional steps:}

\begin{itemize}
    \item[\checkitem] \textbf{Sheets:} Place used sheets in the laundry hamper in the bathroom (no need to make the beds)
    \item[\checkitem] \textbf{Fridge:} Remove all perishable items from the fridge (please help yourself to the takeaway containers in the the kitchen)
    \item[\checkitem] \textbf{Trash:} Empty all bins to outside containers
    \item[\checkitem] \textbf{Dishes:} Wash and put away all used items
    \item[\checkitem] \textbf{Hot tub:} Reset to 80°F (90°F in winter), replace cover
    \item[\checkitem] \textbf{Utilities:} Turn off lights, close windows, set heat to 15°C
    \item[\checkitem] \textbf{Outdoor:} Bring in cushions, cover BBQ
    \item[\checkitem] \textbf{Door:} Press the keypad's lock button and pull firmly on the handle while the door locks; press the button once more---a green light means it's secured
\end{itemize}
